 package com.example.linkRedirector.controller;
 
 import com.example.linkRedirector.model.Poll;
 import com.example.linkRedirector.service.PollService;
 import com.example.linkRedirector.service.PollServiceImpl;
 import jakarta.servlet.http.HttpServlet;
 import jakarta.servlet.http.HttpServletRequest;
 import org.springframework.http.HttpStatus;
 import org.springframework.http.ResponseEntity;
 import org.springframework.web.bind.annotation.*;
 
 import java.util.List;
 
 @RestController
 @CrossOrigin
 @RequestMapping("/api/v1")
 public class PollController {
 
     PollService pollService;
 
     public PollController(PollServiceImpl pollService) {
         this.pollService = pollService;
     }
 
     @PostMapping("polls")
     public ResponseEntity<Poll> createPoll(@RequestBody Poll poll) {
         Poll newPoll = pollService.createPoll(poll);
         return new ResponseEntity<>(newPoll, HttpStatus.CREATED);
     }
 
     @GetMapping("polls/{id}")
     public Poll getPoll(@PathVariable("id") String id) {
         return pollService.getPoll(id);
     }
 
     @GetMapping("polls")
     public List<Poll> getAllPolls() {
         return pollService.getAllPolls();
     }
 
     @PatchMapping("polls/{id}")
     public String updatePoll(@PathVariable("id") String id, @RequestBody Poll poll) {
         return pollService.updatePoll(id, poll);
     }
 
     @DeleteMapping("polls/{id}")
     public String deletePoll(@PathVariable("id") String id) {
         return pollService.deletePoll(id);
     }
 }
